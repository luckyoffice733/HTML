function calculateBMI() {
    const weight = parseFloat(document.getElementById("weight").value);
    const heightCm = parseFloat(document.getElementById("height").value);
    const resultDiv = document.getElementById("result");
    const statusDiv = document.getElementById("weightStatus");
  
    // Reset output
    resultDiv.innerHTML = "";
    statusDiv.innerHTML = "";
  
    // Validate input
    if (isNaN(weight) || weight <= 0 || weight > 640) {
      resultDiv.innerHTML = "Invalid weight.";
      return;
    }
  
    if (isNaN(heightCm) || heightCm <= 0 || heightCm > 200) {
      resultDiv.innerHTML = "Invalid height.";
      return;
    }
  
    const heightM = heightCm / 100;
    const bmi = weight / (heightM * heightM);
    const bmiRounded = bmi.toFixed(2);
  
    let weightStatus = "";
  
    if (bmi < 18.5) {
      weightStatus = "Under Weight";
    } else if (bmi >= 18.5 && bmi <= 24.9) {
      weightStatus = "Healthy Weight";
    } else if (bmi >= 25 && bmi <= 29.9) {
      weightStatus = "Over Weight";
    } else {
      weightStatus = "Obesity";
    }
  
    resultDiv.innerHTML = `Your BMI is ${bmiRounded}`;
    statusDiv.innerHTML = `Weight Status: ${weightStatus}`;
  }

function validateLogin() {
    const username = document.getElementById("username").value.trim();
    const password = document.getElementById("password").value.trim();
    const result = document.getElementById("result");
  
    result.style.color = "red";
  
    if (!username || !password) {
      result.textContent = "Please enter both username and password.";
      return;
    }
  
    if (username.length <= 6) {
      result.textContent = "Username must be more than 6 characters.";
      return;
    }
  
    // Regex: at least 1 lowercase, 1 uppercase, 1 digit, and 1 special character
    const passwordRegex = /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[\W_]).+$/;
  
    if (!passwordRegex.test(password)) {
      result.textContent = "Invalid password.";
      return;
    }
  
    result.style.color = "green";
    result.textContent = "Login successful.";
  }

function calculateEMI() {
    const loanAmount = parseFloat(document.getElementById("loanAmount").value);
    const annualInterestRate = parseFloat(document.getElementById("interestRate").value);
    const loanTermMonths = parseInt(document.getElementById("loanTerm").value);
    const resultDiv = document.getElementById("result");
  
    if (isNaN(loanAmount) || isNaN(annualInterestRate) || isNaN(loanTermMonths) || loanAmount <= 0 || loanTermMonths <= 0) {
      resultDiv.textContent = "Please enter valid input values.";
      return;
    }
  
    const monthlyInterestRate = annualInterestRate / 12 / 100;
    const emi = (loanAmount * monthlyInterestRate * Math.pow(1 + monthlyInterestRate, loanTermMonths)) / 
                (Math.pow(1 + monthlyInterestRate, loanTermMonths) - 1);
  
    resultDiv.textContent = `Monthly EMI: â‚¹${emi.toFixed(2)}`;
  }
